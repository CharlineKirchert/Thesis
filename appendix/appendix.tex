\chapter{First appendix}
For example a detail of an algorithm:

\begin{algorithm}
 \KwIn{A formula in CNF}
 \KwOut{SAT or UNSAT}
 
 \lIf {BCP() = CONFLICT}{\Return{UNSAT}}
 \While{True}{
  \lIf{$\lnot$DecideNextBranch()}{\Return SAT}
  \Else{
    \While{BCP() = CONFLICT}{
      backtrackLevel := AnalyzeConflict()\;
      \lIf{backtrackLevel $<$ 0}{\Return UNSAT}
      \lElse{Backtrack(backtrackLevel)}
    }
  }
 }
 \caption{DPLL}\label{alg:dpll}
\end{algorithm}
